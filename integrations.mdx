## Embedding Third-Party Content

Embed charts, videos, or forms from services like YouTube, Google Forms, or Figma directly into your documentation.

```mdx
<Image
  src="https://www.youtube.com/embed/dQw4w9WgXcQ"
  alt="Embedded YouTube video"
/>

<Iframe
  src="https://forms.gle/example"
  title="Embedded Google Form"
  width="100%"
  height="500"
/>
```

Customize embeds by adjusting width, height, and responsive settings in the editor.

## Setting Up Webhooks for Notifications

Webhooks notify external services about events like document updates or new versions.

### Webhook Events

| Event | Description |
|-------|-------------|
| `doc.updated` | Document content changes |
| `doc.published` | New version published |
| `user.invited` | Team member added |

<Steps>
  <Step title="Create Webhook" icon="plus">
    Navigate to Settings > Webhooks and click New Webhook.
  </Step>
  <Step title="Configure Endpoint" icon="link">
    Enter your endpoint URL, e.g., `https://your-webhook-url.com/webhook`.
  </Step>
  <Step title="Select Events" icon="bell">
    Choose events and save.
  </Step>
</Steps>

<CodeGroup tabs="Node.js,Python">
  ```javascript
  const express = require('express');
  const app = express();
  app.use(express.json());

  app.post('/webhook', (req, res) => {
    console.log('Event:', req.body.event);
    console.log('Data:', req.body.data);
    res.status(200).send('OK');
  });

  app.listen(3000);
  ```
  ```python
  from flask import Flask, request

  app = Flask(__name__)

  @app.route('/webhook', methods=['POST'])
  def webhook():
      event = request.json['event']
      data = request.json['data']
      print(f"Event: {event}")
      print(f"Data: {data}")
      return 'OK', 200

  if __name__ == '__main__':
      app.run(port=3000)
  ```
</CodeGroup>

<Response tabs="Success,Error">
  ```json
  {
    "event": "doc.updated",
    "data": {
      "docId": "doc_123",
      "version": "1.2",
      "timestamp": "2024-10-15T10:00:00Z"
    }
  }
  ```

  ```json
  {
    "error": "Invalid signature",
    "message": "Webhook signature mismatch"
  }
  ```
</Response>

## API Access for Custom Extensions

Access Harsh Savani's API at `https://api.example.com/v1` to build custom tools.

<ParamField path="docId" param-type="string" required="true">
  Unique document identifier.
</ParamField>

<ParamField header="Authorization" param-type="string" required="true">
  Bearer `{YOUR_API_KEY}`.
</ParamField>

<Request tabs="cURL,JavaScript">
  ```bash
  curl -X GET https://api.example.com/v1/docs/{docId} \
    -H "Authorization: Bearer YOUR_API_KEY"
  ```
  ```javascript
  const response = await fetch('https://api.example.com/v1/docs/{docId}', {
    headers: {
      'Authorization': 'Bearer YOUR_API_KEY'
    }
  });
  const data = await response.json();
  ```
</Request>

<Expandable title="Advanced API Usage" default-open="false">
  Use query parameters for filtering:

  <ParamField query="limit" param-type="number" required="false">
    Maximum results (default: 50).
  </ParamField>
</Expandable>

<Callout kind="alert">
  Store API keys securely and rotate them regularly.
</Callout>
